<script>
const colors = document.querySelectorAll('.color');
const img = document.getElementById('productImg');

const map = {
  "rgb(218, 237, 189)": "product-green.jpg",
  "rgb(232, 148, 134)": "product-red.jpg",
  "rgb(255, 231, 135)": "product-yellow.jpg"
};

colors.forEach(c => {
  c.addEventListener('click', () => {
    const col = getComputedStyle(c).backgroundColor;
    if(map[col]) img.src = map[col];
  });
});

// Slider functionality
document.querySelectorAll('.slider').forEach(slider => {
  const slides = slider.querySelectorAll('.slide');
  let index = 0;

  const show = (i) => {
    slides.forEach(s => s.classList.remove('active'));
    slides[i].classList.add('active');
  };

  slider.querySelector('.left').onclick = () => {
    index = (index - 1 + slides.length) % slides.length;
    show(index);
  };

  slider.querySelector('.right').onclick = () => {
    index = (index + 1) % slides.length;
    show(index);
  };
});
</script>
